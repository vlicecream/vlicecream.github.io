#include "MyFirstObject.generated.h"

 * 
 */
UCLASS(Blueprintable)
class DAY01_API UMyFirstObject : public UObject
{
	GENERATED_BODY()
    
public:
    UMyFirstObject();
    
    UPROPERTY(BlueprintReadWrite)  // 蓝图读写宏 Blueprint(蓝图) - Read(读) - Write(写)
    float MYFloat;
    
    UFUNCTION(BlueprintCallable)  // 蓝图调用宏 Blueprint(蓝图) - Call(调用) - able(有能力的)
    void MyFunction();
};

UE的日志

// Fill out your copyright notice in the Description page of Project Settings.

#pragma once

#include "CoreMinimal.h"
#include "UObject/NoExportTypes.h"
#include "MyFirstObject.generated.h"

/**
 * 
 */
UCLASS(Blueprintable)
class DAY01_API UMyFirstObject : public UObject
{
	GENERATED_BODY()
    
public:
    UMyFirstObject();
    
    UPROPERTY(BlueprintReadWrite, Category = "My Data")  // Category 是分类
    float MyFloat;
    
    UFUNCTION(BlueprintCallable, Category = "My Func")
    void MyFunction();
};
// Fill out your copyright notice in the Description page of Project Settings.


#include "MyFirstObject.h"

UMyFirstObject::UMyFirstObject()
{
    MyFloat = 0.0f;
}

void UMyFirstObject::MyFunction()
{
    UE_LOG(LogTemp, Log, TEXT("hello UE")); 
    UE_LOG(LogTemp, Warning, TEXT("hello UE"));
    UE_LOG(LogTemp, Error, TEXT("hello UE"));
}

/*
	UE_LOG 第一个参数 就是 log类型
				 第二个参数 就是 log级别
				 第三个参数 就是 文本 只不过不是普通字符串 是TEXT()
*/

如何删除自定义的c++类

1. 首先在虚幻里面删除跟这个c++类相关的蓝图等

2. 打开工程项目文件夹
	2.1 source 进入 project名字下的目录 这就是一个个的类 只需要删除自己想删除的 不要乱删
	2.2 Binaries 把这个文件夹 直接干掉
	2.3 右键 uproject 点击 Generate开头的 (看了下 mac没有 windows会有)
	2.4 直接打开 uproject 直接构建他
