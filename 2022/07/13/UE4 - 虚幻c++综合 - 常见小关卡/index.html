机关门

机关门 需要跟 蓝图的 timeline 一起去实现
// Fill out your copyright notiUBoxComponentce in the Description page of Project Settings.

#pragma once

#include "CoreMinimal.h"
#include "GameFramework/Actor.h"
#include "TriggerableDoor.generated.h"

UCLASS()
class GETSTARTED_API ATriggerableDoor : public AActor
{
	GENERATED_BODY()
	
public:	
	// Sets default values for this actor's properties
	ATriggerableDoor();
    
    // 定义门网格组件
    UPROPERTY(VisibleAnywhere, BlueprintReadWrite, Category="My Static Mesh")
    class UStaticMeshComponent* DoorMesh;
    
    // 定义触发器网格组件
    UPROPERTY(VisibleAnywhere, BlueprintReadWrite, Category="My Static Mesh")
    class UStaticMeshComponent* TriggerMesh;
    
    // 定义触发器组件
    UPROPERTY(VisibleAnywhere, BlueprintReadWrite, Category="My Static Mesh")
    class UBoxComponent* TriggerBox;
    
    // 定义 机关 & 门 的初始值位置
    UPROPERTY(BlueprintReadWrite, Category="Triggerable Door|Trigger Properties")
    FVector InitTriggerLocation;
    UPROPERTY(BlueprintReadWrite, Category="Triggerable Door|Door Properties")
    FVector InitDoorLocation;
    
    // 定义 定时器句柄 与 门延迟时间
    FTimerHandle CloseDoorTimerHandle;
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="Triggerable Door|Door Properties")
    float DelayTime;
    
protected:
	// Called when the game starts or when spawned
	virtual void BeginPlay() override;

public:	
	// Called every frame
	virtual void Tick(float DeltaTime) override;
    
    // 触碰机关处理函数
    UFUNCTION()
    void OnOverlapBegin(UPrimitiveComponent* OverlappedComponent, AActor* OtherActor, UPrimitiveComponent* OtherComp, int32 OtherBodyIndex, bool bFromSweep, const FHitResult& SweepResult);
    
    // 离开机关处理函数
    UFUNCTION()
    void OnOverlapEnd(UPrimitiveComponent* OverlappedComponent, AActor* OtherActor, UPrimitiveComponent* OtherComp, int32 OtherBodyIndex);
    
    // 机关下降的事件 BlueprintImplementableevent 的意思就是 蓝图中可实现事件 所以这个不再是函数 是事件 所以 也不需要在 cpp 中写
    UFUNCTION(BlueprintImplementableevent, Category="Triggerable Door|Trigger Switch")
    void RaiseTrigger();
    
    // 机关复原的事件 BlueprintImplementableevent 的意思就是 蓝图中可实现事件 所以这个不再是函数 是事件 所以 也不需要在 cpp 中写
    UFUNCTION(BlueprintImplementableevent, Category="Triggerable Door|Trigger Switch")
    void LowerTrigger();
    
    // 打开门事件 BlueprintImplementableevent 的意思就是 蓝图中可实现事件 所以这个不再是函数 是事件 所以 也不需要在 cpp 中写
    UFUNCTION(BlueprintImplementableevent, Category="Triggerable Door|Door Switch")
    void OpenDoor();
    
    // 关闭门事件 BlueprintImplementableevent 的意思就是 蓝图中可实现事件 所以这个不再是函数 是事件 所以 也不需要在 cpp 中写
    UFUNCTION(BlueprintImplementableevent, Category="Triggerable Door|Door Switch")
    void CloseDoor();
    
    // 更新 机关状态(位置) 函数
    UFUNCTION(BlueprintCallable, Category="Triggerable Door|Trigger Switch")
    void UpdateTriggerLocation(FVector Offset);
    
    // 更新 门状态(位置) 函数
    UFUNCTION(BlueprintCallable, Category="Triggerable Door|Door Switch")
    void UpdateDoorLocation(FVector Offset);
};

// Fill out your copyright notice in the Description page of Project Settings.


#include "GamePlay/TriggerableDoor.h"
#include "Components/StaticMeshComponent.h"
#include "Components/BoxComponent.h"
#include "MainPlayer.h"
#include "TimerManager.h"

// Sets default values
ATriggerableDoor::ATriggerableDoor()
{
 	// Set this actor to call Tick() every frame.  You can turn this off to improve performance if you don't need it.
	PrimaryActorTick.bCanEverTick = true;
    
    // 初始化触发器组件(并设置为根组件)
    TriggerBox = CreateDefaultSubobject<UBoxComponent>(TEXT("Box Mesh"));
    RootComponent = TriggerBox;
    /*
        对触发器进行
            1. 大小的设置
            2. 设置体积的状态为仅查询 (状态例如 是否碰撞 查询)
            3. 指定 TriggerBox 自己的类型为世界静态
            4. 让他对所有的频道忽略
            5. 让他只响应 "pawn"频道 的overlap
    */
    TriggerBox->SetBoxExtent(FVector(60.0f, 60.0f, 30.0f));
    TriggerBox->SetCollisionEnabled(ECollisionEnabled::QueryOnly);
    TriggerBox->SetCollisionObjectType(ECollisionChannel::ECC_WorldStatic);
    TriggerBox->SetCollisionResponseToAllChannels(ECollisionResponse::ECR_Ignore);
    TriggerBox->SetCollisionResponseToChannel(ECollisionChannel::ECC_Pawn, ECollisionResponse::ECR_Overlap);
    
    // 初始化 门 & 触发机关(附加到根组件上)
    DoorMesh = CreateDefaultSubobject<UStaticMeshComponent>(TEXT("Door Mesh"));
    DoorMesh->SetupAttachment(GetRootComponent());
    TriggerMesh = CreateDefaultSubobject<UStaticMeshComponent>(TEXT("Trigger Mesh"));
    TriggerMesh->SetupAttachment(GetRootComponent());
    
    // 初始化 门延迟时间
    DelayTime = 1.5f;
}

// Called when the game starts or when spawned
void ATriggerableDoor::BeginPlay()
{
	Super::BeginPlay();
    
    // 去设置 踩到机关 & 松开机关的 触发事件
    TriggerBox->OnComponentBeginOverlap.AddDynamic(this, &ATriggerableDoor::OnOverlapBegin);
    TriggerBox->OnComponentEndOverlap.AddDynamic(this, &ATriggerableDoor::OnOverlapEnd);
    
    
    // 在开始的时候 直接获取到 机关 & 门 位置
    InitTriggerLocation = TriggerMesh->GetComponentLocation();
    InitDoorLocation = DoorMesh->GetComponentLocation();
}

// Called every frame
void ATriggerableDoor::Tick(float DeltaTime)
{
	Super::Tick(DeltaTime);

}

void ATriggerableDoor::OnOverlapBegin(UPrimitiveComponent* OverlappedComponent, AActor* OtherActor, UPrimitiveComponent* OtherComp, int32 OtherBodyIndex, bool bFromSweep, const FHitResult& SweepResult)
{
    // AActor* 这个参数 会把踩到机关的对象直接传输给我们
    // 验证踩到机关的是不是 mainplayer (这里用的 cast 转换方法验证)
    AMainPlayer* MainPlayer = Cast<AMainPlayer>(OtherActor);
    if (MainPlayer) {
        // 踩上去就关闭计时器
        GetWorldTimerManager().ClearTimer(CloseDoorTimerHandle);
        
        LowerTrigger();
        OpenDoor();
    }
}

void ATriggerableDoor::OnOverlapEnd(UPrimitiveComponent* OverlappedComponent, AActor* OtherActor, UPrimitiveComponent* OtherComp, int32 OtherBodyIndex)
{
    // AActor* 这个参数 会把离开机关的对象直接传输给我们
    // 验证离开机关的是不是 mainplayer (这里用的 cast 转换方法验证)
    AMainPlayer* MainPlayer = Cast<AMainPlayer>(OtherActor);
    if (MainPlayer) {
        
        RaiseTrigger();
        
        // 获得当前时间的管理者
        // 这是 用 匿名函数 写的格式   GetWorldTimerManager().SetTimer(CloseDoorTimerHandle, FTimerDelegate::CreateLambda(匿名函数name), DelayTime, false);
        GetWorldTimerManager().SetTimer(CloseDoorTimerHandle, this, &ATriggerableDoor::CloseDoor, DelayTime);
    }
}

void ATriggerableDoor::UpdateTriggerLocation(FVector Offset)
{
    // 在机关移动的时候 会有偏移量 然后直接拿新的位置 改动原本位置
    FVector NewLocation = InitTriggerLocation + Offset;
    TriggerMesh->SetWorldLocation(NewLocation);
}

void ATriggerableDoor::UpdateDoorLocation(FVector Offset)
{
    // 在门移动的时候 会有偏移量 然后直接拿新的位置 改动原本位置
    FVector NewLocation = InitDoorLocation + Offset;
    DoorMesh->SetWorldLocation(NewLocation);
}

物体的随机生成

// Fill out your copyright notice in the Description page of Project Settings.

#pragma once

// #include "CoreMinimal.h"
#include "GameFramework/Actor.h"
#include "SpawnVolume.generated.h"

UCLASS()
class GETSTARTED_API ASpawnVolume : public AActor
{
	GENERATED_BODY()
	
public:	
	// Sets default values for this actor's properties
	ASpawnVolume();

    // 定义 box
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly)
    class UBoxComponent* SpawnBox;
    
    // 在这一块区域 生成的 模型所属类 (只要是所属这个类的 都可以随机生成出来)
    UPROPERTY(EditAnywhere, BlueprintReadOnly, Category="Spawn Volume")
    TArray<TSubclassOf<AActor>> SpawnActorClassesArray;
    
protected:
	// Called when the game starts or when spawned
	virtual void BeginPlay() override;

public:	
	// Called every frame
	virtual void Tick(float DeltaTime) override;

    // 获得区域点 "BlueprintPure" 是 纯净 函数  可以 理解成 你问一个问题 他就给你返回答案
    UFUNCTION(BlueprintPure, Category="Spawn Volume")
    FVector GetSpawnPoint();
    
    // 获得 TSubclassOf<AActor> "BlueprintPure" 是 纯净 函数  可以 理解成 你问一个问题 他就给你返回答案
    UFUNCTION(BlueprintPure, Category="Spawn Volume")
    TSubclassOf<AActor> GetSpawnActorClass();
    
    /*
        "BlueprintNativeEvent" 一般 与 "BlueprintCallable" 一块使用
        这里需要定义两个函数 一个 蓝图使用 一个 c++ 使用
        多出来的c++函数名 比如这个 就要定义 “SpawnActor_Implementation”
    */
    UFUNCTION(BlueprintNativeEvent, BlueprintCallable, Category="Spawn Volume")
    // void SpawnActor(UClass* SpawnClass, FVector SpawnLocation);
    void SpawnActor(UClass* SpawnClass, FVector SpawnLocation);
    void SpawnActor_Implementation(UClass* SpawnClass, FVector SpawnLocation);
};

// Fill out your copyright notice in the Description page of Project Settings.


#include "GamePlay/SpawnVolume.h"
#include "Components/BoxComponent.h"
#include "Kismet/KismetMathLibrary.h"

// Sets default values
ASpawnVolume::ASpawnVolume()
{
 	// Set this actor to call Tick() every frame.  You can turn this off to improve performance if you don't need it.
	PrimaryActorTick.bCanEverTick = true;

    // 初始化 box 并设置为 根组件
    SpawnBox = CreateDefaultSubobject<UBoxComponent>(TEXT("SpawnBox"));
    RootComponent = SpawnBox;
}

// Called when the game starts or when spawned
void ASpawnVolume::BeginPlay()
{
	Super::BeginPlay();

}

// Called every frame
void ASpawnVolume::Tick(float DeltaTime)
{
	Super::Tick(DeltaTime);

}

FVector ASpawnVolume::GetSpawnPoint()
{
    // 获取 box 的位置信息 & 缩放过后的 边界信息
    const FVector Origin = SpawnBox->GetComponentLocation();
    const FVector Extent = SpawnBox->GetScaledBoxExtent();
    // 能够在范围 产生随机的点
    return UKismetMathLibrary::RandomPointInBoundingBox(Origin, Extent);
}

TSubclassOf<AActor> ASpawnVolume::GetSpawnActorClass()
{
    // 在数组中 随机的弄出模板
    if (SpawnActorClassesArray.Num() > 0) {
        const int index = FMath::RandRange(0, SpawnActorClassesArray.Num() - 1);
        return SpawnActorClassesArray[index];
    } else {
        return nullptr;
    }
}

void ASpawnVolume::SpawnActor_Implementation(UClass* SpawnClass, FVector SpawnLocation)
{
    if (SpawnClass) {
        GetWorld()->SpawnActor<AActor>(SpawnClass, SpawnLocation, FRotator(0.0f));
    }
}

// Fill out your copyright notice in the Description page of Project Settings.

#pragma once

#include "CoreMinimal.h"
#include "GameFramework/Actor.h"
#include "FloatingPaltform.generated.h"

UCLASS()
class GETSTARTED_API AFloatingPaltform : public AActor
{
	GENERATED_BODY()
	
public:	
	// Sets default values for this actor's properties
	AFloatingPaltform();
    
    // StaticMesh 组件
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly)
    class UStaticMeshComponent* PlatformMesh;
    
    // 起点
    UPROPERTY(VisibleAnywhere, Category="Floating Platform")
    FVector StartPoint;
    
    // 终点  MakeEditWidget 这个的作用就是 可以在模型中 单独拖动 然后调节到你想要的 EndPoint 但是移动后 是相对距离的哦～ 相对模型初始位置
    UPROPERTY(EditAnyWhere, Category="Floating Platform", meta=(MakeEditWidget = "true"))
    FVector EndPoint;
    
    // 平台移动的速度
    UPROPERTY(EditAnyWhere, Category="Floating Platform")
    float InterpSpeed;
    
    // 平台 移动到目标位置 延迟时间
    UPROPERTY(EditAnyWhere, Category="Floating Platform")
    float DelayTime;
    
    // 是否正在移动中
    bool bInterping;
    
    // 计时器句柄
    FTimerHandle InterpTimerHandle;
    
    // 距离
    float Distance;
    
protected:
	// Called when the game starts or when spawned
	virtual void BeginPlay() override;

public:	
	// Called every frame
	virtual void Tick(float DeltaTime) override;

};
// Fill out your copyright notice in the Description page of Project Settings.


#include "GamePlay/FloatingPaltform.h"
#include "Components/StaticMeshComponent.h"

// Sets default values
AFloatingPaltform::AFloatingPaltform()
{
 	// Set this actor to call Tick() every frame.  You can turn this off to improve performance if you don't need it.
	PrimaryActorTick.bCanEverTick = true;
    
    // StaticMesh 组件
    PlatformMesh = CreateDefaultSubobject<UStaticMeshComponent>(TEXT("PlatformMesh"));
    RootComponent = PlatformMesh;
    
    // 初始化 起点 & 终点 & 移动速度
    StartPoint = FVector(0.0f);
    EndPoint = FVector(0.0f);
    InterpSpeed = 200.0f;
    
    // 初始化 延迟 & 移动状态
    DelayTime = 2.0f;
    bInterping = true;
}

// Called when the game starts or when spawned
void AFloatingPaltform::BeginPlay()
{
	Super::BeginPlay();
	
    // 获取开始的时候位置
    StartPoint = GetActorLocation();
    // 这里因为 "EndPoint" 是局部坐标 演示一下 怎么换回 世界坐标
    EndPoint += StartPoint;
    
    // 获得 起始 结束 之间的距离
    Distance = (EndPoint - StartPoint).Size();
}

// Called every frame
void AFloatingPaltform::Tick(float DeltaTime)
{
	Super::Tick(DeltaTime);
    
    // 如果正在移动状态
    if (bInterping) {
        // 获取到 actor 的位置
        const FVector CurrentLocation = GetActorLocation();
        // 进行 数学的 位置加法 "VInterpTo" 这个是 移动会很快 最后会很慢
        // const FVector NewLocation = FMath::VInterpTo(CurrentLocation, EndPoint, DeltaTime, InterpSpeed);
        // "VInterpConstantTo" 则是匀速
        const FVector NewLocation = FMath::VInterpConstantTo(CurrentLocation, EndPoint, DeltaTime, InterpSpeed);
        // 改变坐标
        SetActorLocation(NewLocation);
        
        // 到了最终位置 获取当前 与 起始的距离
        float NowDistance = (GetActorLocation() - StartPoint).Size();
        if (Distance - NowDistance <= 0.1f) {
            // lambda 表达式
            auto ToggleInterpState = [this]() {
                bInterping = !bInterping;
            };
            ToggleInterpState();
            
            // 开启计时器
            GetWorldTimerManager().SetTimer(InterpTimerHandle, FTimerDelegate::CreateLambda(ToggleInterpState), DelayTime, false);
            
            // 移动并开启计时器完毕 将起始位置和最终位置进行 交换
            FVector TempVec = StartPoint;
            StartPoint = EndPoint;
            EndPoint = TempVec;
        }
    }
}
