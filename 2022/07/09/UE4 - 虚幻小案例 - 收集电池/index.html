UE4 小案例 收集电池

场景迁移

我们在 epic game 商城 搜索内容示例 下下来 
  1. 建一个内容示例的项目
 	2. 建一个第三人称的项目 c++
  (内容示例项目)
  3. 然后 点击 content 搜索 BP_DemoRoom & SM_Battery_Medium & P_electricity_arc 
  4. 找到后 右键 -> Asset Actions -> Migrate 全部选取 然后 找到自己的项目 导入进去
  (收集电池项目)
  5. 确认导入成功 新建一个关卡 在content 建Maps文件夹 然后关卡保存到Maps
  6. 将 demoroom 放入关卡 调整好 Player Start
  (结束)

// Sets default values
ASpawnVolume::ASpawnVolume()
{
 	// Set this actor to call Tick() every frame.  You can turn this off to improve performance if you don't need it.
	PrimaryActorTick.bCanEverTick = true;
    
    // 创建触发器的盒子组件 & 设置为根组件
    WhereToSpawn = CreateDefaultSubobject<UBoxComponent>(TEXT("WhereToSpawn"));
    RootComponent = WhereToSpawn;
    
    // 初始化时间的延迟区间
    SpawnDelayRangeLow = 1.0f;
    SpawnDelayRangeHigh = 4.5f;
}

// 获取随机点
FVector ASpawnVolume::GetRandomPointInVolume()
{
    // 初始位置
    FVector SpawnOrigin = WhereToSpawn->Bounds.Origin;
  
    // 结束位置
    FVector SpawnExtent = WhereToSpawn->Bounds.BoxExtent;
  
    return UKismetMathLibrary::RandomPointInBoundingBox(SpawnOrigin, SpawnExtent);
}

void ASpawnVolume::SpawnPickup()
{
    if (WhatToSpawn != NULL) {
        UWorld* const World = GetWorld();
        
        if (World) {
            // 设置触发参数
            FActorSpawnParameters SpawnParams;
            SpawnParams.Owner = this;
            SpawnParams.Instigator = GetInstigator();
            
            // 获取触发的随机位置
            FVector SpawnLocation=GetRandomPointInVolume();
            
            // 获取触发物体的随机旋转
            FRotator SpawnRotation;
            SpawnRotation.Yaw = FMath::FRand() * 360.0f; // y轴
            SpawnRotation.Pitch = FMath::FRand() * 360.0f; // z轴
            SpawnRotation.Roll = FMath::FRand() * 360.0f; // x轴
            
            // 触发拾取
            ACollectActor* const SpawnedPickup = World->SpawnActor<ACollectActor>(WhatToSpawn, SpawnLocation, SpawnRotation, SpawnParams);
            
            // 延迟的设置
            SpawnDelay = FMath::FRandRange(SpawnDelayRangeLow, SpawnDelayRangeHigh);
            // 绑定到全局计时器
            GetWorldTimerManager().SetTimer(SpawnTimer, this, &ASpawnVolume::SpawnPickup, SpawnDelay, false);
            
        }
    }
}

// Called when the game starts or when spawned

	Super::BeginPlay();

    GetWorldTimerManager().SetTimer(SpawnTimer, this, &ASpawnVolume::SpawnPickup, SpawnDelay, false);
}

	Super::Tick(DeltaTime);

ABatterCollectExampleCharacter::ABatterCollectExampleCharacter()
{
	// Set size for collision capsule
	GetCapsuleComponent()->InitCapsuleSize(42.f, 96.0f);

	// set our turn rates for input
	BaseTurnRate = 45.f;
	BaseLookUpRate = 45.f;

	// Don't rotate when the controller rotates. Let that just affect the camera.
	bUseControllerRotationPitch = false;
	bUseControllerRotationYaw = false;
	bUseControllerRotationRoll = false;

	// Configure character movement
	GetCharacterMovement()->bOrientRotationToMovement = true; // Character moves in the direction of input...	
	GetCharacterMovement()->RotationRate = FRotator(0.0f, 540.0f, 0.0f); // ...at this rotation rate
	GetCharacterMovement()->JumpZVelocity = 600.f;
	GetCharacterMovement()->AirControl = 0.2f;

	// Create a camera boom (pulls in towards the player if there is a collision)
	CameraBoom = CreateDefaultSubobject<USpringArmComponent>(TEXT("CameraBoom"));
	CameraBoom->SetupAttachment(RootComponent);
	CameraBoom->TargetArmLength = 300.0f; // The camera follows at this distance behind the character	
	CameraBoom->bUsePawnControlRotation = true; // Rotate the arm based on the controller

	// Create a follow camera
	FollowCamera = CreateDefaultSubobject<UCameraComponent>(TEXT("FollowCamera"));
	FollowCamera->SetupAttachment(CameraBoom, USpringArmComponent::SocketName); // Attach the camera to the end of the boom and let the boom adjust to match the controller orientation
	FollowCamera->bUsePawnControlRotation = false; // Camera does not rotate relative to arm
    
    // 创建球形组件 & 绑定到 根组件 & 设置 球形范围
    CollectionSphere = CreateDefaultSubobject<USphereComponent>(TEXT("CollectionSphere"));
    CollectionSphere->SetupAttachment(RootComponent);
    CollectionSphere->SetSphereRadius(200.0f);
    
    // 初始化人物能量值 & 刚开场的实际能量 等于 初始能量
    InitialPower = 2000.0f;
    CharacterPower = InitialPower;

    
    // 设置两种速度
    SpeedFactor = 0.75f;
    BaseSpeed = 10;
    
	// Note: The skeletal mesh and anim blueprint references on the Mesh component (inherited from Character) 
	// are set in the derived blueprint asset named MyCharacter (to avoid direct content references in C++)
}

// Input

void ABatterCollectExampleCharacter::SetupPlayerInputComponent(class UInputComponent* PlayerInputComponent)
{
	// Set up gameplay key bindings
	check(PlayerInputComponent);
	PlayerInputComponent->BindAction("Jump", IE_Pressed, this, &ACharacter::Jump);
	PlayerInputComponent->BindAction("Jump", IE_Released, this, &ACharacter::StopJumping);

    /*
     Action 是绑定动画 & 短促绑定
     Axis 是绑定轴向 & 持续绑定
     这里我们绑定收集事件
     */
    PlayerInputComponent->BindAction("Collect", IE_Pressed, this, &ABatterCollectExampleCharacter::CollectPickups);
    
	PlayerInputComponent->BindAxis("MoveForward", this, &ABatterCollectExampleCharacter::MoveForward);
	PlayerInputComponent->BindAxis("MoveRight", this, &ABatterCollectExampleCharacter::MoveRight);

	// We have 2 versions of the rotation bindings to handle different kinds of devices differently
	// "turn" handles devices that provide an absolute delta, such as a mouse.
	// "turnrate" is for devices that we choose to treat as a rate of change, such as an analog joystick
	PlayerInputComponent->BindAxis("Turn", this, &APawn::AddControllerYawInput);
	PlayerInputComponent->BindAxis("TurnRate", this, &ABatterCollectExampleCharacter::TurnAtRate);
	PlayerInputComponent->BindAxis("LookUp", this, &APawn::AddControllerPitchInput);
	PlayerInputComponent->BindAxis("LookUpRate", this, &ABatterCollectExampleCharacter::LookUpAtRate);

	// handle touch devices
	PlayerInputComponent->BindTouch(IE_Pressed, this, &ABatterCollectExampleCharacter::TouchStarted);
	PlayerInputComponent->BindTouch(IE_Released, this, &ABatterCollectExampleCharacter::TouchStopped);

	// VR headset functionality
	PlayerInputComponent->BindAction("ResetVR", IE_Pressed, this, &ABatterCollectExampleCharacter::OnResetVR);
}

void ABatterCollectExampleCharacter::MoveForward(float Value)
{
	if ((Controller != nullptr) && (Value != 0.0f))
	{
		// find out which way is forward
		const FRotator Rotation = Controller->GetControlRotation();
		const FRotator YawRotation(0, Rotation.Yaw, 0);

		// get forward vector
		const FVector Direction = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::X);
		AddMovementInput(Direction, Value);
	}
}

void ABatterCollectExampleCharacter::MoveRight(float Value)
{
	if ( (Controller != nullptr) && (Value != 0.0f) )
	{
		// find out which way is right
		const FRotator Rotation = Controller->GetControlRotation();
		const FRotator YawRotation(0, Rotation.Yaw, 0);
	
		// get right vector 
		const FVector Direction = FRotationMatrix(YawRotation).GetUnitAxis(EAxis::Y);
		// add movement in that direction
		AddMovementInput(Direction, Value);
	}
}

// 按下按键 收集圆形范围内的CollectActor
void ABatterCollectExampleCharacter::CollectPickups()
{
    // 定义一个数组
    TArray<AActor*> CollectedActors;
    
    // 遍历所有区域覆盖的Actor 存入数组
    CollectionSphere->GetOverlappingActors(CollectedActors);
    
    // 记录收集的电池能量
    float CollectPower = 0.0f;
    
    // 遍历数组
    for (int32 iCollected = 0; iCollected < CollectedActors.Num(); ++iCollected)
    {
        // 转化Actor为ACollectActor
        ACollectActor* const TestPickup = Cast<ACollectActor>(CollectedActors[iCollected]);
        
        // 检查是否转化成功 && 是否被销毁 && 是否激活状态
        if (TestPickup && !TestPickup->IsPendingKill() && TestPickup->GetActive())
        {
            // 收集pickup
            TestPickup->WasCollected();
            
            // 检查CollectActor是否是电池类
            ABatteeryCollect* const TestBattery = Cast<ABatteeryCollect>(TestPickup);
            if (TestBattery)
            {
                CollectPower += TestBattery->GetPower();
            }
            
            // 检查状态
            TestPickup->SetActive(false);
        }
    }
    // 如果 结束 角色实际能量值大于0 就把能量值添加上去
    if (CollectPower > 0.0f)
    {
        UpdatePower(CollectPower);
    }
}

// 获得初始能量值
float ABatterCollectExampleCharacter::GetInitialPower()
{
    return InitialPower;
}

// 获得实际能量值
float ABatterCollectExampleCharacter::GetCurrentPower()
{
    return CharacterPower;
}

// 更新能量值 (吸收电池能量)
void ABatterCollectExampleCharacter::UpdatePower(float PowerChange)
{
    CharacterPower += PowerChange;
    
    // 获取角色的移动
    GetCharacterMovement()->MaxWalkSpeed = BaseSpeed + SpeedFactor * CharacterPower;
    
    // 调用蓝图实现方法
    PowerChangeEffect();
}

 * 
 */
UCLASS()
class BATTERCOLLECTEXAMPLE_API ABatteeryCollect : public ACollectActor
{
	GENERATED_BODY()
	
public:
    // 获取能量
    float GetPower();

    // 构造函数
    ABatteeryCollect();
    
    // 重写父类虚函数
    void WasCollected_Implementation() override;
    
protected:
    
    UPROPERTY(EditAnyWhere, BlueprintReadWrite, Category="Power", meta=(BlueprintProtect="true"))
    float BatteryPower;
};

// Fill out your copyright notice in the Description page of Project Settings.


#include "BatteeryCollect.h"

ABatteeryCollect::ABatteeryCollect()
{
    // 设置物理状态
    GetMesh()->SetSimulatePhysics(true);
    
    // 初始化电池能量值
    BatteryPower = 150.0f;
}

void ABatteeryCollect::WasCollected_Implementation()
{
    // 调用父类方法
    Super::WasCollected_Implementation();
    
    // 销毁电池
    Destroy();
}

// 返回电池能量值
float ABatteeryCollect::GetPower()
{
    return BatteryPower;
}
