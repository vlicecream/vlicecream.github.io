#pragma once

#include "CoreMinimal.h"
#include "GetStartedGameModeBase.h"
#include "Level1GameMode.generated.h"

/**
 * 
 */
UCLASS()
class GETSTARTED_API ALevel1GameMode : public AGetStartedGameModeBase
{
	GENERATED_BODY()
	
public:
    // 主UI生成
    UPROPERTY(EditAnywhere, BlueprintReadOnly, Category="UI Widget")
    TSubclassOf<class UUserWidget> MainUIClass;
    
    // 把对象放在指针 方便我们去读取
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Category="UI Widget")
    UUserWidget* MainUI;
    
protected:
    // 重写beginplay方法
    virtual void BeginPlay() override;
};

// Fill out your copyright notice in the Description page of Project Settings.


#include "GameLogic/Level1GameMode.h"
#include "Blueprint/UserWidget.h"
#include "Kismet/GameplayStatics.h"

void ALevel1GameMode::BeginPlay()
{
    // 先调用父类方法
    Super::BeginPlay();
    
    if (MainUIClass) {
        // 第一个参数 哪一个物体拥有它(player) 第二个参数 生成哪一个UI
        MainUI = CreateWidget<UUserWidget>(UGameplayStatics::GetPlayerController(this, 0), MainUIClass);
        
        // 添加到视口
        if (MainUI) {
            MainUI->AddToViewport();
        }
    }
}
