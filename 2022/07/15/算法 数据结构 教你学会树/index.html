/*
	递归的思路 - 前序遍历
		我们可以 preorder(root) 表示遍历到哪一个节点
		拿出当前节点的val 随后 "->left" & "->right" 进入 左子树 & 右子树
		直到检测出空node 直接return 即可
		时间复杂度 O(n) 空间复杂度O(n)
*/ 
class Solution {
public:
    vector<int> preorderTraversal(TreeNode* root) {
        vector<int> res;
        preorder(root, res);
        return res;
    }

    void preorder(TreeNode* root, vector<int>& res) {
        if (root == nullptr) return;
        res.push_back(root->val);
        preorder(root->left, res);
        preorder(root->right, res);
    }
};

/*
	迭代的思路 - 前序遍历
		迭代的思路 跟 递归大致相同 
		区别就是 递归会隐形的调用栈 我们迭代将这个栈表示出来
		时间复杂度 O(n) 空间复杂度O(n)
*/
class Solution {
public:
    vector<int> preorderTraversal(TreeNode* root) {
        vector<int> res;
        if (root == nullptr) {
            return res;
        }

        stack<TreeNode*> stk;
        TreeNode* node = root;
        while (!stk.empty() || node != nullptr) {
            while (node != nullptr) {
                res.emplace_back(node->val);
                stk.emplace(node);
                node = node->left;
            }
            node = stk.top();
            stk.pop();
            node = node->right;
        }
        return res;
    }
};

/*
	Morris迭代
		Morris迭代的核心思路就是利用树的大量空闲指针，实现空间的减少
		1. 新建临时节点 node
		2. 如果左子节点为nullptr 将当前节点加入答案 并遍历右子节点
		3. 如果左子节点不为nullptr 将当前节点的左子树中找到当前节点在中序遍历下的前驱节点
			3.1 如果前驱节点的右子节点为空，将前驱节点的右子节点设置为当前节点
					然后将当前节点加入答案，并将前驱节点的右子节点更新为当前节点。当前节点更新为当前节点的左子节点
			3.2 如果前驱节点的右子节点为当前节点，将它的右子节点重新设为空。当前节点更新为当前节点的右子节
		时间复杂度 O(n) 空间复杂度 O(1)
*/

class Solution {
public:
    vector<int> preorderTraversal(TreeNode *root) {
        vector<int> res;
        if (root == nullptr) {
            return res;
        }

        TreeNode *p1 = root, *p2 = nullptr;

        while (p1 != nullptr) {
            p2 = p1->left;
            if (p2 != nullptr) {
                while (p2->right != nullptr && p2->right != p1) {
                    p2 = p2->right;
                }
                if (p2->right == nullptr) {
                    res.emplace_back(p1->val);
                    p2->right = p1;
                    p1 = p1->left;
                    continue;
                } else {
                    p2->right = nullptr;
                }
            } else {
                res.emplace_back(p1->val);
            }
            p1 = p1->right;
        }
        return res;
    }
};

例题1 - 二叉树的最大深度

https://leetcode.cn/problems/maximum-depth-of-binary-tree/

/*
	这道题其实很简单
	我们又可以用深度优先搜索 也可以用广度优先搜索
*/

/*
	我们先来看 深度优先搜索
		我们先假想我们知道了左子树和右子树的 随后最大深度 就是 max(leftDepth, rightDepth) + 1
		随后就是解决 leftDepth & rightDepth 的问题了
		左子树与右子树的计算方式其实是一样的 
			
		具体而言，在计算当前二叉树的最大深度时，可以先递归计算出其左子树和右子树的最大深度
		然后在 O(1)O(1) 时间内计算出当前二叉树的最大深度。递归在访问到空节点时退出
		
		时间复杂度 O(n) 空间复杂度(height) 这里的 height 指的是 二叉树的深度
*/

class Solution {
public:
    int maxDepth(TreeNode* root) {
        if (root == nullptr) return 0;
        return max(maxDepth(root->left), maxDepth(root->right)) + 1;
    }
};
好了 我们其实已经解决掉这个题目了
但是算法这个玩意 终究是 坚持下来的活～ 量变引起质变
所以我们不能只只限于写出来 所以 我们要去想想 怎么用广度优先搜索去写出来 并看看 大佬的写法
  
/*
	首先 广度优先搜索 都是用 queue 去实现的
		这里 我们需要当前层 node 去放入 queue 
		随后对 queue 所有的 node 去做扩展
		拿出 ans 去维护 深度
*/
  
class Solution {
public:
    int maxDepth(TreeNode* root) {
        if (!root) return 0;
        queue<TreeNode*> q;
        q.push(root);
        int ans = 0;
        while (!q.empty()) {
            int len = q.size();
            while (len > 0) {
                TreeNode* node = q.front(); q.pop();
                if (node->left) q.push(node->left);
                if (node->right) q.push(node->right);
                len--;
            }
            ans++;
        }
        return ans;
    }
};

例题2 - 验证二叉搜索树

/*
	这道题是验证 这个树 是不是二叉搜索树
		看我的另一篇讲 树 的理论基本知识的 盆友 绝对记得 中序遍历的一个性质
		那就是 只要是二叉搜索树 中序遍历下来 绝对是升序！
		
		so？
*/

class Solution {
public:
    bool isValidBST(TreeNode* root) {
        vector<int> res;
        inorder(root, res);
        for (int i = 1; i < size(res); ++i) {
            if (res[i] <= res[i-1]) return false;
        }
        return true;
    }

    void inorder(TreeNode* root, vector<int>& res) {
        if (!root) return;
        inorder(root->left, res);
        res.push_back(root->val);
        inorder(root->right, res); 
    }
};
