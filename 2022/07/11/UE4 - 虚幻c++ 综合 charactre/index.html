void AMainPlayer::TurnAtRate(float Rate)
{
    // 因为Rate要么是1，0，-1 不可能说靠这个转动 太慢了 所以得要 * 一个基数 而且要注意帧率平衡
    float Value = Rate * BaseTurnRate * GetWorld()->GetDeltaSeconds();
    if (Value != 0.0f) {
        AddControllerYawInput(Value);
    }
}

{
    // 获取到Mainplayer
    PlayerRef = Cast<AMainPlayer>(TryGetPawnOwner());
}

void UMainPlayerAnimInstance::UpdateAnimationProperties()
{
    // 做个安全判断 防止空指针
    if (PlayerRef == nullptr) PlayerRef = Cast<AMainPlayer>(TryGetPawnOwner());
    if (PlayerRef) {
        FVector SpeedVector = PlayerRef->GetVelocity();
        FVector PlanarSpeed = FVector(SpeedVector.X, SpeedVector.Y, 0.0f);
        Speed = PlanarSpeed.Size();
    }
    
    bIsInAir = PlayerRef->GetMovementComponent()->IsFalling();
}

```cpp
// Fill out your copyright notice in the Description page of Project Settings.

#pragma once

#include "CoreMinimal.h"
#include "GameFramework/Character.h"
#include "MainPlayer.generated.h"

// 角色移动状态
UENUM(BlueprintType)
enum class EPlayerMovementStatus : uint8
{
    EPMS_Normal UMETA(DisplayName = "Normal"),
    EPMS_Sprinting UMETA(DisplayName = "Sprinting"),
    EPMS_Dead UMETA(DisplayName = "Dead")
};

// 角色耐力状态
UENUM(BlueprintType)
enum class EPlayerStaminaStatus : uint8
{
    EPSS_Normal UMETA(DisplayName = "Normal"),
    EPSS_Exhausted UMETA(DisplayName = "Exhausted"),  // 精疲力竭的
    EPSS_ExhausedRecovering UMETA(DisplayName = "ExhausedRecovering")  // 精疲力竭的恢复状态
};

UCLASS()
class GETSTARTED_API AMainPlayer : public ACharacter
{
	GENERATED_BODY()

public:
	// Sets default values for this character's properties
	AMainPlayer();
    
    
    // 创建 悬臂 & 相机 组件
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Category="SpringArmComponent");
    class USpringArmComponent* SpringArm;
    UPROPERTY(VisibleAnywhere, BlueprintReadOnly, Category="CameraComponent");
    class UCameraComponent* FollowCamera;
    
    // 转向速率
    float BaseTurnRate;
    float BaseLookUpRate;
    
    // 人物的 血条 & 耐力值 & 金币
    UPROPERTY(EditDefaultsOnly, BlueprintReadOnly, Category="Player Status");
    float MaxHealth;
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="Player Status")
    float Health;
    UPROPERTY(EditDefaultsOnly, BlueprintReadOnly, Category="Player Status");
    float MaxStamina;
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="Player Status")
    float Stamina;
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="Player Status")
    int Coins;
    
    // 人物冲刺相关(1. 冲刺速度 2. 3. 耐力下降速度 4. 耐力疲劳区 5. 耐力状态 6. 移动状态 7. 是否按下冲刺键 8. )
    UPROPERTY(EditAnywhere, BlueprintReadOnly, Category="Player Status")
    float RunningSpeed;
    
    UPROPERTY(EditAnywhere, BlueprintReadOnly, Category="Player Status")
    float SprintingSpeed;
    
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="Player Status")
    float StaminaConsumeRate;
    
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="Player Status", meta=(ClampMin = 0, ClampMax = 1, UIMin = 0, UIMax = 1))
    float ExhaustedStaminaRatio;
    
    UPROPERTY(VisibleAnywhere, BlueprintReadWrite, Category="Player Status")
    EPlayerStaminaStatus StaminaStatus;
    
    UPROPERTY(VisibleAnywhere, BlueprintReadWrite, Category="Player Status")
    EPlayerMovementStatus MovementStatus;
    
    bool bLeftShiftKeyDown;
    
    
protected:
	// Called when the game starts or when spawned
	virtual void BeginPlay() override;

public:	
	// Called every frame
	virtual void Tick(float DeltaTime) override;

	// Called to bind functionality to input
	virtual void SetupPlayerInputComponent(class UInputComponent* PlayerInputComponent) override;
    
    // jump函数可以继承父类的jump();
    virtual void Jump() override;
    
    // 绑定按键映射函数
    void MoveForward(float Value);
    void MoveRight(float Value);
    
    // 鼠标映射函数
    void Turn(float Value);
    void LookUp(float Value);
    
    // 键盘 手柄 up down left right 映射函数
    void TurnAtRate(float Rate);
    void LookUpAtRate(float Rate);
    
    // 增加血量(拾取血瓶)
    UFUNCTION(BlueprintCallable)
    void IncreaseHealth(float value);
    // 增加耐力(拾取蓝瓶)
    UFUNCTION(BlueprintCallable)
    void IncreaseStamina(float value);
    // 增加金币(拾取金币)
    UFUNCTION(BlueprintCallable)
    void IncreaseCoin(float value);
    
    // 受伤处理函数
    virtual float TakeDamage(float Damage, struct FDamageEvent const& DamageEvent, AController* EventInstigator, AActor* DamageCauser) override;
    
    // 按下(抬起)冲刺键位 改变键位状态
    FORCEINLINE void LeftShiftKeyDown() { bLeftShiftKeyDown = true; };
    FORCEINLINE void LeftShiftKeyUp() { bLeftShiftKeyDown = false; };
};

void AMainPlayer::LookUp(float Value)

    if (GetControlRotation().Pitch < 270.0f && GetControlRotation().Pitch > 180.0f && Value > 0.0f) {
        return;
    }
    if (GetControlRotation().Pitch < 180.0f && GetControlRotation().Pitch > 45 && Value < 0.0f) {

void AMainPlayer::TurnAtRate(float Rate)
{
    // 因为Rate要么是1，0，-1 不可能说靠这个转动 太慢了 所以得要 * 一个基数 而且要注意帧率平衡
    float Value = Rate * BaseTurnRate * GetWorld()->GetDeltaSeconds();
    if (Value != 0.0f) {
        AddControllerYawInput(Value);
    }
}

void AMainPlayer::LookUpAtRate(float Rate)
{
    // 因为Rate要么是1，0，-1 不可能说靠这个转动 太慢了 所以得要 * 一个基数 而且要注意帧率平衡
    float Value = Rate * BaseLookUpRate * GetWorld()->GetDeltaSeconds();
    if (GetControlRotation().Pitch < 270.0f && GetControlRotation().Pitch > 180.0f && Value > 0.0f) {
        return;
    }
    if (GetControlRotation().Pitch < 180.0f && GetControlRotation().Pitch > 45 && Value < 0.0f) {
        return;
    }
    AddControllerPitchInput(Value);
}

void AMainPlayer::IncreaseHealth(float value)
{
    // 因为血是有限制的 所以要用到夹值
    Health = FMath::Clamp(Health + value, 0.0f, MaxHealth);
}

void AMainPlayer::IncreaseStamina(float value)
{
    // 因为耐力是有限制的 所以要用到夹值
    Stamina = FMath::Clamp(Stamina + value, 0.0f, MaxStamina);
}

void AMainPlayer::IncreaseCoin(float value)
{
    Coins += value;
}

float AMainPlayer::TakeDamage(float Damage, struct FDamageEvent const& DamageEvent, AController* EventInstigator, AActor* DamageCauser)
{
    if (Health - Damage <= 0.0f) {
        Health = FMath::Clamp(Health - Damage, 0.0f, MaxHealth);
    } else {
        Health -= Damage;
    }
    return Health;
}
