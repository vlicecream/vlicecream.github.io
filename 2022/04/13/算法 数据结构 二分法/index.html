// https://leetcode-cn.com/problems/search-insert-position/ 搜索插入位置
// 搜索插入位置
func searchInsert(nums []int, target int) int {
  left, right, ans := 0, len(nums) -1, len(nums)
  for left <= right {
    mid := (right + left)/2
    if nums[mid] >= target {
      ans = mid
      right = mid - 1
    } else {
      left = mid + 1
    }
  }
  return ans
}
