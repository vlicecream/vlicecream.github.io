c++ 引用-reference

reference

int x = 0;
int* p = &x; // p is a pointer to x
int& p = x; // r is a reference to x
int x2 = 5;

r = x2;
int& r2= r;
/*
	int* p = &x;
	int *p = &x;
	以上两行都是可以的 但是个人更喜欢是int* p
	因为这样的话 p 就是一个变量  这个变量是什么类型 是 pointer to int
*/
/*
  int* 是 pointer to int
  int& 是 reference to int
  比如 int* p = &x; 我们会说 指针指向了x
  而 int& r = x; 我们会说 r 代表了 x 其实 r 里面有一个指针指向了 x
  r 只要代表了 x 就不可以代表别的变量 pointer不同 pointer还可以指向别的变量
*/
/*
	所以我们接下来看
	int x2 = 5;
	r = x2;
	这两行代码 r 不能重新代表其他的变量了 所以 经过 "r = x2;" 后 r 和 x 都变成了 5
	再然后 int r2 = r; 现在r2是5 ("r2" 代表 "r" 亦相当于代表了 "x")
*/

假象

/*
	我们观察 "x" 跟 "r" 的地址和大小 会发现
	1. sizeof(r) = sizeof(x)
  2. &x == &r
  object 和其 reference 的大小相同 地址也相同(但是都是假象)
  java里面的所有变量都是reference
*/

reference的常见用途

/*
	reference 就是一种漂亮的pointer
	reference 通常不用于声明变量 而用于参数类型和返回类型的描述
*/ 

// 声明
void func1(Cls* pobj) { pobj->xxx(); } // pass by pointer
void func2(Cls obj) { obj.xxx(); } // pass by value
void func3(Cls& obj) { obj.xxx(); } // pass by reference
// 调用
Cls obj;
func1(&obj);
func2(obj);
func(obj);

// 我们能发现 pass by pointer 跟其他的都不一样 而 "value" 和 "reference" 被调用端 写法相同 调用端接口也相同

// 这也就是我们最好用reference 不行再value

签名

// 以下被视为 "same signature" 所以两者不能同时存在
double image(const double& im) { ... }
double image(const double im) { ... }

/*
	签名的部分为
		imag(const double im)
	说白了就是不含 return type
	
	这两行 如果不算相同的签名 那么 imag(obj) 就不知道要调用谁
*/
// 问题就来了 const 算不算函数签名呢
