c++ 重载-new delete

global的重载

void* myAlloc(size_t size) { return malloc(size); }
void* myFree(void* ptr) { return free(ptr); }

// 他们不可以被声明于一个namespace中
inline void* operator new(size_t size) { cout << "jjhou global new() \n"; return myAlloc(size); }
inline void* operator new[](size_t size) { cout << "jjhou global new[]() \n"; return myAlloc(size); }
inline void* operator delete(void* ptr) { cout << "jjhou global delete() \n"; return myFree(ptr); }
inline void* operator delete[](void* ptr) { cout << "jjhou global delete[]() \n"; return myFree(ptr); }

成员函数的重载

class Foo {
public:
  void* operator new(size_t);
  void* operator new[](size_t);
  void operator delete(void*, size_t);
  void operator delete[](void*, size_t);
}

重载new( ) delete( )

/*
	我们可以重载 class member operator new() 写出多个版本 前提是每一个版本的声明都必须有独特的参数列
	其中第一参数必须是size_t 其余参数以new所指定的placement arguments 为初值
	出现在new(...) 小括号内的便是所谓的placement arguments
*/

/*
	我们也可以重载 class member operator delete() 写出多个版本
	但他们绝不会被delete调用 只有当new所调用的ctor抛出exception 才会调用这些重载版的operator delete()
	他只可能这样被调用
	主要处理还未能完全创建成功的object所占用的memory
*/
// new()示例

class Foo {
public:
  Foo() { cout << "Foo::Foo()" << endl; }
  Foo(int) { cout << "Foo:Foo(int)" << endl; throw Bad(); } // 故意在这抛出错误 测试delete
  
  // 这就是一般的 operator new()
  void* operator new(size_t size) { return malloc(size); }
  // 这就是标准库已经提供的 placement new() 的重载 ( 这里简化 )
  void* operator new(size_t size, void* start) { return start; }
  // 这个才是崭新的 placement new
  void* operator new(size_t size, long extra) { return mallc(size+extra); }
  // 这又是一个placement new
  void* operator new(size_t size, long extra, char init) { return malloc(size_extra); }
// 这是搭配上述的new 的各个 placement delete
// 当ctor发出异常 这儿对应的opertor (placement) delete 就会被调用
// 用途就是释放对应的placement new 分配的memory
// 即使 operator delete(...)未能一一对应也不会出现任何报错 这样的意思是 放弃处理ctor所发出的异常

// 这个就是一般的opertor delete() 的重载
void operator delete(void*, size_t) { cout << "operator delete(void*, size_t)" << endl; }
// 这个对应第二个
void operator delete(void*, void*) { cout << "operator delete(void*, void*)" << endl; }
// 对应第三个
void operator delete(void*, long) { cout << "operator delete(void*, long)" << endl; }
// 对应第四个
void operator delete(void*, long, char) { cout << "operator delete(void*, long, char)" << endl; }

private:
	int m_i;
};
