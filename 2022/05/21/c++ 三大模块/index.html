c++ 三个小知识点

数量不定的模板参数

// 就是python 和 golang 的可变长参数

template <typename T, typename... Types>
void print (const T& firstArg, const Types&.. args) {
  cout << firstArg << endl;
  print(args...);
}

语法糖

/*
	python 也有语法糖哦～ 糖很甜 吃糖的时候能有幸福感
  所以语法糖就是能让你感到很幸福 
  对于程序员 很幸福的事情是什么了 当然是秃头(不是)
  进入秃头 啊不是 进入主题
  语法糖就是能让你少写代码
  怎么样 是很幸福吧
*/ 
list<string> c;
...
list<string>::iterator ite;
ite = find(c.begin(), c.end(), target);

// 这段代码就可以用 "auto" 这个语法糖来优化
list<string> c;
...
auto ite = find(c.begin(), c.end(), target);
// auto 可以让编译器帮我们 获取到这个变量的类型 那我们是不是都用auto就好了呢

/*
	答案肯定不是的 因为c++是一个能帮助我们写出很好的代码规范的一个高级语言 我们不能太依赖于auto 而且全部用auto 有时候也会报错的
*/

list<string> c;
...
auto ite;
ite = find(c.begin(), c.end(), target);

// 这样子的就是错误的

for循环

// for循环 有很多种写法 
for (int i = 1; i < size(vector); ++i) {
  ...
}
for ( dec1 : coll ) {
  ...
}

/*
	这种写法 ":" 后面跟的是容器 会将容器里面的值 一个个的遍历出来 放到 dec1 里面
*/
for (int i : { 2, 3, 4, 5, 6, 7}) {
  cout << i >> endl;
}

/*
	这种写法也是 会把 { } 内的元素 放到 i 中
*/
vector<double> vec;
...
for ( auto elem : vec ) {
  cout << elem << endl;
}

// auto 获取elem 的类型 然后 遍历 vec 里面的元素 放到elem 里面 这种的是 pass by value

for (auto& elem : vec ) {
  elem *= 3;
}

// 这种的是 pass by reference
