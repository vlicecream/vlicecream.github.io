c++ 头文件 操作符重载

操作符重载

c2 += c1;
/*
	+= 就是一个操作符 那么操作符是怎么被编译器看待的 += 是二元操作符
		+= 会用到左边的c2上，如果c2有对+=的定义 就会调用+=函数 操作符本来也就是一个函数
*/
inline complex& __doapl(complex* ths, const complex& r) {
  ths->re += r.re;
  ths->im += r.im;
  return *ths; // 返回的是一个value
}

inline complex& complex::operator+=(const complex& r) {
  return __doapl(this, r);
}

/*
	这个代码就是重写了 += 函数 多包装了一层
*/

成员函数this

在上述代码其实更深一层看 是这样的
inline complex& complex::operator+=(this, const complex& r) {
  return __doapl(this, r);
}
// 这个this 就是操作符左边的 也就是c2 但是在我们代码里面不可以写出来 知道他的存在就好
// 那么this 一定在第一个参数吗 其实是不一定的 但是不用考虑

Return by reference语法分析

inline complex& __doapl(complex* ths, const complex& r) {
  //  ...
  return *ths;
}
/*
  complex *ths 是指针的方式传进来
  return 的时候 *ths 是返回指针所指向的value
  然而声明返回值的时候 说返回的是一个内存地址&
	这是允许的 你返回一个value 接受端怎么接受 是不需要管的
*/

传递者无需知道接收者是以referebce形式接收

class body 外的各种定义

inline double imag(const complex& x) {
  return x.imag();
}

inline double real(const complex& x) {
  return x.real();
}
{
  complex c1(2, 1);
  cout << imag(c1);
  cout << real(c1);
}

非成员函数

{
  complex c1(2, 1);
  complex c2;
	
  c2 = c1 + c2;
  c2 = c1 + 5;
  c2 = 7 + c1;
}

// 下面三个操作符 “+” 由复数和实部组成三种 所以需要有三种全局函数
inline complex operator + (const complex& x, const complex& y) {
  return complex (real(x) + real(y), imag(x) + imag(y));
}
inline complex operator + (const complex& x, double y) {
  return complex (real(x) + y, imag(x);
}
inline complex operator + (double x, const complex& y) {
  return complex (x + real(y), imag(y));
}

临时对象

上面三个 "+" 函数绝不可return by reference
因为 他们返回的必定是个local object
/*
	上述的 + 并不是+=
	x 和 y 相加后的 data 并没有 地方放 所以在函数里一定会放在一个东西里 在这里也就是complex()
	complex() 相当于 int() 都会创建一个临时对象 这样的意思就是下一行就挂了 只是临时的作用
*/
{
  complex(); // 临时对象 构造函数是又会有默认值的
  complex(4, 5); // 临时对象
}

一个好玩的源码

inline complex operator + (const complex& x) {
  return x;
}

inline complex operator - (const complex& x) {
  return complex (-real(x), -imag(x));
}

/*
	上面 + - 是正负的意思 "-" 就是取反 产生新的数据 所以需要临时对象 不能return by reference
	但是 "+" 为什么不 return by reference 呢 这是标准库里面的源码 但是源码不代表就是完美的
	这里其实个人理解 return by reference 更佳
	所以一定要有自己的想法
*/

{
  complex c1(2 ,1);
  complex c2;
  cout << -c1;
  cout << +c2;
}
