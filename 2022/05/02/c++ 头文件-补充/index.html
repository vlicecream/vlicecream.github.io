// conplex 分为 data func 两个部分 只要你加入static 就是静态的意思
complex c1, c2, c3;
cout << c1.real();
cout << c2.real();
// 上述代码其实要是看透本质等于下面代码
complex c1, c2, c3;
cout << complex::real(&c1);
cout << complex::real(&c2);

// 静态用在哪
存储静态的数据 不会变动的
class Accounr {
public:
  static double my_rate;
  static void set_rate(const double &x);
};
double Account::my_rate = 8.0; // 静态的数据 都要写这个 至于赋值 随便


int main() {
  Account::set_rate(5.0);
  
  Account a;
  a.set_rate(7.0);
}

把ctors放在private区

// 我们可以把ctors放在私有区域 可以放一个唯一的自己 然后外界通过一个static function来得到唯一的自己

class A {
public:
  static A &getInstance( return a; );
  setup() { ... }
private:
  A();
  A(const A &rhs);
  static A a;
  ...
};

// 这个其实就是单例模式 但是这个代码不太好 因为唯一的a一直存在 可以放在函数里进行改善
