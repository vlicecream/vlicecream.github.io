Stack

/*
	存在于scope(作用域)的一块内存空间
	当你调用函数 函数本身即会形成一个stack用来放置他所接受的参数 以及返回地址
	在函数本体内声明的任何变量 其所使用的内存块都取自上述的stack
*/

Heap

/*
	由操作系统提供的一块golbal内存空间 程序可动态分配从某中获得若干区块
*/

stack objects 的生命周期

// 在scope结束之际结束 就会调用他的析构函数
// 这种作用域内的object 又称为 auto object 因为会被自动清理
class Complex { ... };
...
{
  static Complex c2(1, 2); 
}

// c2 便是所谓的static object 生命在作用域scope 结束之后仍然存在 直到整个程序结束

  ...
}

Heap objects 生命周期

class Complex { ... };
...
{
  complex* p = new complex;
  ...
  delete p;
}

/*
	p 所指的便是heap object 其生命在他被deleted之际结束
*/
class Complex { ... };
...
{
  complex* p = new complex;
}

/*
	这会出现内存泄漏
	当作用域结束后 heap object还存在 然后指针p已经死了 所以没有办法 delete
*/

new

// new: 先分配memory 再调用ctor
  
complex* pc = new Complex(1, 2);

// 编译器转换为
Complex* pc;
void *mem = operator new( sizeof(complex) );  // 分配内存 其内部调用malloc(n)
pc = static_cast<Complex*>(mem);  //转型
pc->Complex::Complex(1, 2);

delete

// 先调用dtor 在释放memory

delete ps;

// 编译器转换为
String::~String(ps); // 析构函数
operator delete(ps); // 释放内存
