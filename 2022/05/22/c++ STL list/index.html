c++ STL list深度解析

list源代码(G2.9)

/*
	list 是环状双向链表
*/
// G2.9 的 list 源代码

template <class T, class Alloc = alloc>
class list {
protected:
	typedef __list_node<T> list_node;
public:
  typedef list_node* link_type
  typedef __list_iterator<T, T&, T*> iterator;
protected:
  link_type node;
...
};

template <class T>
struct __list_node {
  typedef void* void_pointer;
  void_pointer prev;
  void_pointer next;
  T data;
};

template <class T, class Ref, class Ptr>
struct __list_iterator {
  typedef T value_type;
  typedef Ptr pointer;
  typedef Ref reference;
...
};

/*
	首先在此说明这是G2.9 list 的部分源代码
	然后我们来看 __list_node的实现
		他有 "data" 数据部分 有 "next" 下一个指针 有 "prev" 上一个指针
		但是 这两个指针 确实 void pointer
		这很不好 因为 链表最适合的就是指向自己 但是在G2.9源码确实是这么写的
	然后我们看 list 的数据部分 "node"
		node 是 link_type 类型 然后link_type 是 list_node pointer 所以list数据是4byte(32位系统)
*/

list’s iterator

template <class T, class Ref, class Ptr>
struct __list_iterator {
  typedef T value_type;
  typedef Ptr pointer;
  typedef Ref reference;
...
};

/*
	随后我们来看 iterator
	链表是非连续空间 所以iterator不能是指针 但是又要有指针的功能
		所以 想到了智能指针 iterator++ 要去看目前的next指针然后跳到下一个
		所以 iterator 必须是 class
*/
template<class T, class Ref, class Ptr>
struct __list_iterator {
  typedef __list_iterator<T, Ref, Ptr> self;
  typedef bidirectional_iterator_tag iterator_category; // 1
  typedef T value_type; // 2
  typedef Ptr pointer; // 3
  typedef Ref reference; // 4
  typedef __list_node<T>* link_type;
  typedef ptrdiff_t difference_type; // 5
  
  link_type node;
  
  reference operator*() const { return (*node).data; }
  poniter operator->() const { return &(operator*()); }
  self& operator++() { node = (link_type)((*node).next); return *this; }
  self operator++(int) { self tmp = *this; ++*this; return tmp; }
  ...
};

/*
	每个容器 至少要有上面标注的 5个typedef
	随后因为要做成 class 智能指针 所以要操作符重载很多东西
	接下来详细介绍 ++
*/
self operator++(int) { self tmp = *this; ++*this; return tmp; } 
self& operator++() { node = (link_type)((*node).next); return *this; }

/*
	由于 "++" 很特殊 因为可以 "++i" or "i++" 
		所以c++语法规定 在 前置"++" 的括号里 不要写参数 后置“++” 要写
		熟悉操作符重载的小伙伴知道这个是无实际意义的 因为不管写不写 都会把obj"i" 给传进行 所以只是区分前置后置
	前置"++" 的写法是
		将(*node).next的指针去赋予前面的node 
		也就是将自己的next指针给node 这样node就会指向下一个元素
	后置"++"的写法是
		这里有一个点 我们要记住 有很多操作符 这些操作符很有可能重载
		self tmp = *this 我们可能会以为操作符重载* 但是其实不是 因为 "=" 也被重载 这是个拷贝构造
		以此类推
	然后我们看两个 “++” 的返回类型
		前置 是return by reference 后置是 return by value
		这是为什么 其实是像偶像致敬 哈哈
*/

int i(6);
++++i;  // ++(++i)  正确
i++++;  // (i++)++  错误
/*
	我们是不能后置两次 ++ 的 但是可以前置两次 所以这里才会出现 return by reference(value)的区别 因为 int 就是这么处理的
*/
T& operator*() const { return (*node).data; }
T* operator->() const { return &(operator*()); }

// 第一行是直接取当前node指向的data数据
// 第二行就是取数据的内存地址
