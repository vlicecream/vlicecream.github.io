c++ STL 分配器

malloc()

/*
	我们通过operator new() 只要是这个 就会调用 malloc()
	那么malloc()的角色就是分配内存
	具体的我们会在内存管理里面讲 
	
	目前 需要知道的 melloc 分配的内存是要比你所要的大小大了不少 大在cookie 上下都有
	所以你要的比例越大 附加的东西越小 反之附加的东西则越大
*/

分配器allocators

// VC6 对allocator的实现
template<class _Ty> {
public:
  typedef _SIZE size_type;
  typedef _PDFT difference_type;
  typedef _Ty _FARQ *pointer;
  typedef _Ty value_type;
  pointer allocate(size_type _N, const void *) { return (_Allicate((difference_type)_N, (pointer)0)); }
  void deallocate(boid _FARQ *_P, size_type) { operator delete(_P); }
};

// 其中_Allocate()定义如下

template<class _Ty> inline _Ty _FARQ *_Allocate(_PDFT _N, _Ty _FARQ *) {
  if (_N < 0) _N = 0;
  return ((_Ty _FARQ *) operator new((_SIZT)_N * sizeof (_Ty)));
}

/*
	VC6+ 的 allocator只是以 ::operatpr new 和 ::operator delete 完成allocate() 和 deallocate()
	没有特殊设计
*/

// 分配 512 ints
int* p = allocator<int>().allocate(512, (int*)0);
allocator<int>().deallocate(p, 512);

/*
	这种直接() 就是临时变量 下一行代码就会死去
	然后我们调用allocate 需要分配512 内存 ，第二个参数没有固定 就直接传0
	最后我们调用deallocate 释放内存 同样也要把512输入进去 在我们开发的时候不会专门去记多少内存 只会去记指针
	所以不推荐我们直接调用分配器 但是容器没有这个困扰
	
	而且有个大问题 分配的内存小 开销的比例大 就不可以接受 因为说到底还是 直接使用 c 的 melloc() 和 free()
*/
/*
	BC++ 的 allocator只是以 ::operatpr new 和 ::operator delete 完成allocate() 和 deallocate()
	G2.9 的设计也是同样的 但是这三个的问题也都一样没有解决 都是直接使用 melloc() 和 free()
	G2.9 不同的就是写了一个注释
	精华就是 不要使用这个文件 这个头文件没有被任何的include 用的是另一个allocators
*/
// G2.9 STL 对allocator的使用
template <class T, class Alloc = alloc>
class vector {
  ...
};

// vector list deque map set 用的默认分配器都是alloc

// alloc 的核心思想 就是减少 melloc 的使用 因为melloc()注定会带来额外的开销

// 但是在G4.9 其实默认分配器的换掉了alloc alloc还是存在的 改名为__pool_alloc 也可以用这个分配器

vector<string, __gnu_css::__pool_alloc<string>> vec;
