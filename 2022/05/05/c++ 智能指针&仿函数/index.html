c++ 智能指针&仿函数

pointer-like classes

// 智能指针 一个类很像指针 拥有指针的功能确还能额外多做点事 这就叫智能指针
template<class T>
class shared_ptr {
public:
	T& operator *() const { return *px; }
+	T& operator ->() const { return px; }
	shared_ptr(T* p) : px(p) { } // 构造函数
private:
	T* px;
	long* pn;
};
struct Foo {
    ...
    void method(void) { ... };
};
shared_ptr<Foo> sp(new Foo);
Foo f(*sp);
sp -> method;

/*
	在这里new了一个sp对象 调用shared_ptr构造函数 没有问题
	随后Foo f(*sp) 操作符重载 return *px 取到指针所指向的值 也没有问题
	再然后的 sp -> 这里的意思 是通过sp这个对象 调用method方法 实际也就是 px->method;
	-> 这个符号很特别 会一直作用下去 -> 作用于sp 虽然作用完了 但是 还是能px->method 所以体现了这一点
*/

迭代器

// 迭代器也有智能指针
template<class T, class Ref, class Ptr>
struct __list_iterator {
  typedef __list_node<T>* link_type;
  link_type node;
  bool operator==(const self& x) const { return node == x.node };
  bool operator->() const { return &(operator*()); }
  reference operator*() const { return (*node).data; }
  pointer operator->() const { return &(operator*()); }
}
// 像前面的智能指针 就是普通用法 而迭代器的这个例子则就是体现了更好的利用指针去搞事情

function-like classes 仿函数

template <class T>
struct identity : public unary_function<T, T> {
    const T& operator() (const T& x) const { return x; }
};
template <class Pair>
struct select1st : public unary_function<Pair, typename Pair::first_type> {
    const typename Pair::first_type& operator() (const Pair& x) const { return x.first; }
};
template <class Pair>
struct select2nd : public unary_function<Pair, typename Pair::second_type> {
    const typename Pair::second_type& operator() (const Pair& x) const { return x.second; }
};
// 只要里面有()则他有可能就是仿函数 可以见操作符重载的重要性
