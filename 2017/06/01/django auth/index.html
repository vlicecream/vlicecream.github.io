二、AUTH模块的使用

1. 代码演示
# views
    from django.shortcuts import render, HttpResponse, redirect
    from django.contrib import auth
    from django.contrib.auth.decorators import login_required
    from django.contrib.auth.models import User

    # Create your views here.

    """
    我们再用auth的时候 用就用全套的auth提供的方法
    """


    def form_input(request):
        # 判断是否为POST
        if request.method == 'POST':
            # 获取前端输入内容
            username = request.POST.get('username')
            password = request.POST.get('password')
            """
            验证密码 有两道难题
            1.表在哪
            2.密文怎么对比
            """
            # 直接用auth模块即可，但是要注意必须把两个参数传进去才行
            user_obj = auth.authenticate(username=username, password=password)
            print(user_obj)  # 用户对象 如果上述账号密码不正确则返回一个None
            # 登陆成功即保存用户数据
            auth.login(request, user_obj)  # 为什么不用session方法，因为我们用全套的auth方法，而且用了这个auth方法不管在哪你都能request.user拿到登录用户
            return redirect('/home/')
        return render(request, 'form_input.html')


    @login_required
    def home(request):
        print(request.user)  # 依然是用户对象，如果不登录直接拿就是匿名用户
        # 我们可以用这个内置方法来判断是不是匿名用户
        print(request.user.is_authenticated)
        return HttpResponse('welcome')


    @login_required
    def set_password(request):
        # 检测是否为POST
        if request.method == 'POST':
            # 获取到用户数据
            old_password = request.POST.get('old_password')
            new_password = request.POST.get('new_password')
            confirm_password = request.POST.get('confirm_password')
            # 对比两次密码是否一致
            if new_password == confirm_password:
                # 对比老密码是否正确,这个地方得用新方法
                is_right = request.user.check_password(old_password)
                if is_right:
                    request.user.set_password(new_password)  # 仅仅是在修改对象属性
                    request.user.save()  # 这一步才是真正保存进数据库的一步
                    return HttpResponse('success')
        return render(request, 'set_password.html')


    def logout(request):
        auth.logout(request)
        return redirect('/login/')


    def register(request):
        if request.method == 'POST':
            username = request.POST.get('username')
            password = request.POST.get('password')
            User.objects.create_user(username=username, password=password)
        return render(request, 'form_input.html')

    
# urls
    from django.conf.urls import url
    from django.contrib import admin
    from app01 import views
    urlpatterns = [
        url(r'^admin/', admin.site.urls),
        url(r'^index/', views.form_input),
        url(r'^home/', views.home),
        url(r'^set_password/', views.set_password),
        url(r'^logout/', views.logout),
    	url(r'^register/', views.register),
    ]

三、基于AUTH模块的使用总结方法

1. 对比用户名&密码是否正确
	user_obj = auth.authenticate(request, username=username, password=password)
    # 括号内必须传入用户名和密码
    print(user_obj)
    print(user_obj.username)
    print(user_obj.password)
2. 保存用户状态
	auth.login(request, user_obj)
    # 执行了该方法 无论在哪都能request.user
3. 判断用户当前是否登录
	request.user.is_anthenticated()
4. 获取当前用户
	request.user
5. 校验用户是否登录装饰器
	from django.contrib.auth.decorators import login_required
    # 局部配置
    @login_required(login_url='/login/')
    # 全局配置
    LOGIN_URL = '/LOGIN/'
    """
    1.如果同时有局部和全局，应该听谁的
    	局部>全局
    2.局部和全局那个好呢
    	全局的好处在于不用写重复代码 但是跳转页面都很单一
    	局部的好处在于不同的视图函数在用户没有登录的情况下可以跳转到不同的页面
    """
6. 比对原密码
	request.user.check_password(old_password)
7. 修改密码
	request.user.set_password(new_password)  # 只是改变了对象属性
    request.user.save()  # 这才是修改数据库数据
8. 注销密码
	auth.logout(request)
9. 注册用户
	# 创建普通用户
    User.objects.create_user(username=username, password=password)
    # 创建超级用户
	User.objects.create_superuser(username=username, email='123@qq.com', password=password)
    # 在这里创建超级用户必须要填邮箱，而命令行却不用

四、AUTH模块包拓展

1. 代码演示
    from django.db import models
    from django.contrib.auth.models import User, AbstractUser


    # Create your models here.

    # 第一种方法  一对一表关系 不推荐
    class UserDetail(models.Model):
        phone = models.CharField(max_length=16)
        user = models.OneToOneField(to='User')

    # 第二种方法
    class UserInfo(AbstractUser):
        """
        如果执行了AbstractUser
        那么在执行数据库迁移命令的时候auth_user表就不会再创建出来了
        而UserInfo表中会出现auth_user所有的字段外加自己扩展的阻断
        这么做的好处在于你能够直接点击你自己的表更加快速的完成操作及扩展

        前提:
            1.在继承之前没有执行过数据库迁移命令
                auth_user没有被创建 如果当前库已经创建了那么你就重新换一个库
            2.继承的类里面不要覆盖AbstractUser里面的字段名
                表里面有的字段都不要动，至扩展额外字段即可
            3.需要在配置文件中告诉django你要用UserInfo代替auth_user
                AUTH_USER_MODEL = 'app01.UserInfo'
                                    '应用名.表名'    
        """
        phone = models.BigIntegerField()
